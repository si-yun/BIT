3개 넓이의 총합을 구하는 프로그램을 짜시오


class Shape{
	int getarea() {
		return 0;
	}
}
// 원의 넓이 사각형의 넓이 삼각형의 넓이 
class Circle extends Shape{
	private int radius;
	Circle(int radius){
		this.radius = radius;
	}
	int getarea(){
		return radius*radius*(int)Math.PI;
	}
}
class Triangle extends Shape{
	private int width;
	private int height;
	
	Triangle(int width, int height){
		this.height = height;
		this.width = width;
	}
	int getarea() {
		return height*width/2;
	}
}
class Rectangle extends Shape{
	private int height;
	private int width;
	
	Rectangle(int height, int width){
		this.height = height;
		this.width = width;
	}
	
	int getarea() {
		return height*width;
		
	}
}

public class AreaTest {
	public static void main(String[] args) {
		Circle cir = new Circle(10);
		Triangle tri = new Triangle(20,10);
		Rectangle rec = new Rectangle(10, 20);
		 System.out.println(cir.getarea()+tri.getarea()+rec.getarea());
		 Shape[] shapes = new Shape[3];
		 
		 shapes[0] = new Circle(10);
		 shapes[1] = new Triangle(20, 10);
		 shapes[2] = new Rectangle(10, 20);
		 
		 int totalArea = 0;
		 for(Shape shape :shapes) {
			 totalArea = totalArea + shape.getarea();
			 
		 }
		 System.out.println(totalArea);
	}
}
